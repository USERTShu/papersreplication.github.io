    【发布】小猿口算个位数比较示例程序代码

    2024/10/11

    demo简介

    　　本程序可自动化<<小猿口算>>APP中个位数比较练习的过程。
    　　注意：本程序仅仅是示例程序，功能少且性能差。并且对使用条件有着非常严苛的要求，例如屏幕分辨率、手机投屏到电脑后窗口放置的位置等，程序并未对不同情况做出适配，因此不建议尝试运行该代码。
    　　编写思路是获取要计算的题目的截图，经过OCR识别得到图中的内容，进而计算后在绘制答案区操控鼠标进行绘制。
    　　重申：该demo程序仅展示该思路的可行性，并未对代码本身做出优化，不建议学习该程序的编写风格！

import easyocr

"""

pg.PAUSE = 0.01
mode = 2
gap_time = 2 if mode == 1 else 0.5

def draw(start_pos, end_pos):
    length = ((start_pos[0] - end_pos[0])**2 + (start_pos[1] - end_pos[1])**2) ** 0.5 + 1e-12
    rate = 5
    delta_x = (end_pos[0] - start_pos[0]) / length * rate
    delta_y = (end_pos[1] - start_pos[1]) / length * rate
    x = start_pos[0]; y = start_pos[1]
    pg.mouseDown(*start_pos)
    while abs(x - end_pos[0]) > 5 or abs(y - end_pos[1]) > 5:
        x += delta_x
        y += delta_y
        pg.moveTo(x, y)
    pg.moveTo(*end_pos)
    pg.mouseUp()

def process(text_list) -> tuple[bool, str]:
    # TODO 增加鲁棒性
    if len(text_list) == 0:
        print("Detect Nothing")
        return False, ""
    print(text_list)
    if mode == 1:
        text_list = "".join(text_list)
        text_list = text_list[:-1]
        return True, str(eval(text_list))
    else:
        if eval(text_list[0]) > eval(text_list[-1]):
            return False, "g"
        elif eval(text_list[0]) < eval(text_list[-1]):
            return False, "l"
        elif eval(text_list[0]) == eval(text_list[-1]):
            return False, "e"
        else:
            print("Detect Nothing")
            return False, ""
