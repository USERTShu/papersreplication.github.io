    练习题：二维向量类

    2024/10/09

    要求

    　　我们预计在下周发布几道题目，用于让大家参考自己的python水平是否达到要求，届时仍未达到要求且想加入论文复现组的同学，就需要加快python学习进度了，如果有同学已经学习完了“类”，可以先尝试完成一下该题：
    　　实现这样一个类：Vector2d(二维向量)。其功能为：表示一个二维向量，支持向量加减法运算，求模长运算，求反向量运算，判断两向量是否垂直，是否共线。
    　　为求简便，使用直角坐标系表示。
    　　该类的使用方法示例：
    　　vec_a = Vector2d(3, 5) # 坐标(3, 5)
    　　vec_b = Vector2d(1, 2) # 坐标(1, 2)
    　　print(vec_a + vec_b) # 向量加法 形式1
    　　print(vec_a.add(vec_b)) # 形式2 实现任意一种形式即可


    参考代码

    
# 仅供参考
class Vector2d:

    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __add__(self, other):
        """形式1"""
        return Vector2d(self.x + other.x, self.y + other.y)
        # 或直接打印: print(f"({self.x + other.x}, {self.y + other.y})")
        
    def add(self, other):
        """形式2"""
        return Vector2d(self.x + other.x, self.y + other.y)
        # 或直接打印: print(f"({self.x + other.x}, {self.y + other.y})")
        
    def __sub__(self, other):
        """形式1"""
        return Vector2d(self.x - other.x, self.y - other.y)
        # 或直接打印: print(f"({self.x - other.x}, {self.y - other.y})")

    def sub(self, other):
        """形式2"""
        return Vector2d(self.x - other.x, self.y - other.y)
        # 或直接打印: print(f"({self.x - other.x}, {self.y - other.y})")
        
    def get_length(self):
        """求模"""
        return (self.x**2 + self.y**2) ** 0.5
        
    def get_negative(self):
        """求反"""
        return Vector2d(-self.x, -self.y)
        # 或直接打印: print(f"({-self.x}, {-self.y})")

    def is_parallel(self, other):
        """判断是否共线"""
        return self.x * other.y == self.y * other.x
        
    def is_vertical(self, other):
        """判断是否垂直"""
        return self.x * other.x + self.y * other.y == 0
        
    def __repr__(self):
    
        # 若采用"直接打印"的方式 该__repr__()方法可忽略
        # 建议返回一个新的Vector2d对象 而不是直接打印
        return f"({self.x}, {self.y})"
    
if __name__ == '__main__':
    a = Vector2d(3, 5)
    b = Vector2d(1, 2)
    print("a + b =", a + b)
    print("a - b =", a.sub(b))
    print("a的模长为:", a.get_length())
    print("b的相反向量为:", b.get_negative())
    print("a与b" + ("" if a.is_parallel(b) else "不") + "共线")
    print("a与b" + ("" if a.is_vertical(b) else "不") + "垂直")

    运行结果

    　　a + b = (4, 7)
    　　a - b = (2, 3)
    　　a的模长为: 5.830951894845301
    　　b的相反向量为: (-1, -2)
    　　a与b不共线
    　　a与b不垂直
